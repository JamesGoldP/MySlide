基于jQuery的对象切换插件： myslide 1.0插件 (点击下载)

所有参数：

options = $.extend({
            autoChange: true,       //是否自动切换
            changeTime: 5000,       //自动切换时间
            prevBtn: null,             //上一张幻灯片按钮
            nextBtn: null,             //下一张幻灯片按钮
            haveNum: false,          //是否有数字导航
            linkText: false,            //是否有带透明的文字背景
            effect: "fade",            //效果
            delayTime:300,           //鼠标经过时对象切换迟滞时间，推荐值为300ms
            hoverChange:true,       //鼠标经过是否切换
            hoverChagneClass:"on" //鼠标移上去的样式
        }, options || {});

1- 默认最简易模式

$(".banner3>a").myslide();

//最简易模式不带任何参数，即直接切换选择器选择的对象组，所有参数为默认：隔5秒自动切换，图片过渡效果持续1秒

2- 带前后按钮

$('.banner4>a').myslide({
      prevBtn:"#prev4",
      nextBtn:"#next4"
});

//带前后按钮

3- 使用数字导航切换及自定义当前状态数字的class

$('.banner5>a').myslide({
      haveNum:true,
      hoverChagneClass:"act"
});

//hoverChagneClass用来自定义移上去的class

4- 切换对象为其他，这里为包含图片和标题的层

$('.banner6>a').myslide({
      haveNum:true,
      hoverChagneClass:"act",
      linkText:true
});

//

5- 小图片切换

$('.banner7>a').myslide({
      haveNum:true,       hoverChagneClass:"act"
});

//如果想换成图片,直接加一个class为slide_ul就行了。

6- 更换效果

$('.banner8>a').myslide({
      haveNum:true,
      hoverChagneClass:"act",
      effect:"slide",
      hoverChange:false
});

//有fade和slide 两种方式

6- 只有透明文字背景

$('.banner9>a').myslide({
});

//有fade和slide 两种方式
